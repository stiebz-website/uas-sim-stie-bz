<script>
   function submitExam() {
        // 1. Ambil Data Mahasiswa
        const nama = document.getElementById('nama').value;
        const nim = document.getElementById('nim').value;

        if(!nama || !nim) {
            alert("Mohon isi Nama dan NIM terlebih dahulu!");
            return;
        }

        // 2. Konfirmasi
        const confirmSubmit = confirm("Apakah Anda yakin ingin menyelesaikan ujian? Jawaban akan diunduh ke dalam file .txt");
        if (!confirmSubmit) return;

        // 3. Kumpulkan semua jawaban
        let content = "========================================\n";
        content += "LEMBAR JAWABAN UAS SIM\n";
        content += "========================================\n";
        content += `Nama : ${nama}\n`;
        content += `NIM  : ${nim}\n`;
        content += `Waktu Submit : ${new Date().toLocaleString()}\n`;
        content += "========================================\n\n";

        // Loop untuk mengambil jawaban soal no 1 sampai 10
        for (let i = 1; i <= 10; i++) {
            const answer = document.getElementsByName(`q${i}`)[0].value;
            content += `--- SOAL NO ${i} ---\n`;
            content += `${answer ? answer : "[TIDAK DIJAWAB]"}\n\n`;
        }

        // 4. Proses Pembuatan File Download (.txt)
        const blob = new Blob([content], { type: "text/plain;charset=utf-8" });
        const url = URL.createObjectURL(blob);
        
        // Membuat elemen link sementara untuk men-trigger download
        const link = document.createElement("a");
        link.href = url;
        link.download = `UAS_SIM_${nama}_${nim}.txt`; // Nama file otomatis
        document.body.appendChild(link);
        link.click();
        
        // Bersihkan
        document.body.removeChild(link);
        
        alert("Jawaban berhasil diunduh! Silakan kirim file .txt tersebut ke Dosen.");
    }
</script>
